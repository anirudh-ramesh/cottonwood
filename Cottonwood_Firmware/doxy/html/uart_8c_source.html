199 void Serial_SendBufferedChar(u8 ch)
200 {
201  if (conSerTxInProgress)
202  { /* Waiting for tx done is lazy, i.e. it is done at the next character
203  after a flush */
204  conSerTxInProgress = 0;
205  conSerIdx = 0;
206  while ( check_transmitted_ != AS399X_CMD_TRANSMITTED );
207  }
208 
209  conSerArray[conSerIdx] = ch;
210  conSerIdx++;
211 
212  if(conSerIdx >= sizeof(conSerArray) || ch=='\n')
213  { /* if buffer is full or we send a newline */
214  sendArrayN(conSerArray, conSerIdx);
215  conSerTxInProgress = 1;
216  }
217 }

233 void CON_print(void* format,...)
234 {
235  uint32 i=0;
236  uint32 argint=0;
237  uint32 PercentageFound=FALSE,ZeroFound=FALSE;
238  uint32 NoOfNonzeroChars,NoOfZeros;
239  uint32 NoOfChars = 0;
240  uint32 upperCase = FALSE; /*user ABCDEF of abcdef for hex digits */
241  uint8 argchar=0;
242  uint8 CharSent[NO_OF_NIBBLES + 1];
243  s16 hs;
244 
245  va_list argptr;
246  va_start(argptr,format);
247 
248  while (*((uint8*)format) != NULL_CHARACTER)
249  {
250  switch (*((uint8*)format))
251  {
252  case '%':
253  hs = 0;
254  if (PercentageFound == TRUE)
255  {
256  PercentageFound=FALSE;
257  CON_putchar('%');
258  }
259  else
260  {
261  PercentageFound=TRUE;
262  }
263 
264  break;
265  case 'h':
266  if (PercentageFound)
267  {
268  hs++;
269  }
270  else
271  {
272  CON_putchar(*((uint8*)format));
273  }
274  break;
275  case 'X':
276  upperCase = TRUE;
277  /* fall through */
278  case 'c':
279  case 'l':
280  case 'D':
281  case 'U':
282  case 'd':
283  case 'u':
284  case 'x':
285 
286  /* Initialise the counters to 0 */
287  /*NoOfChars=0;*/
288  NoOfNonzeroChars=0;
289  NoOfZeros=0;
290 
291  /*if d/x was preceeded by % it represents some format */
292  if (PercentageFound == TRUE)
293  {
294  int nibbles = NO_OF_NIBBLES;
295  if ( hs == 2)
296  {
297  nibbles = 2;
298  argint=(u8)va_arg(argptr,u8);
299  }
300  else if ( hs == 1) argint=(unsigned short)va_arg(argptr,unsigned short);
301  else argint=(s16)va_arg(argptr,int);
302 
303  /*shift and take nibble by nibble from MSB to LSB*/
304  for (i=nibbles; i > 0; i--)
305  {
306  CharSent[i-1]=(uint8)((argint & ((0xf)<<(4*(nibbles-1)))) >> (4*(nibbles-1)));
307  if ((CharSent[i-1] >= 0x0a) && (CharSent[i-1] <= 0x0f))
308  {
309  /*if the nibble is greater than 9 */
310  CharSent[i-1] = (uint8)
311  ( (uint32)CharSent[i-1]
312  + (upperCase ? ALPHA_UC_TO_INTEGER
313  : ALPHA_LC_TO_INTEGER)
314  );
315  }
316  else
317  {
318  /*if nibble lies from 0 to 9*/
319  CharSent[i-1]=(uint8)((uint32)CharSent[i-1]+NUMCHAR_TO_INTEGER);
320  }
321  /*see how many nibbles the number is actually filling*/
322  if (CharSent[i-1] == '0')
323  {
324  if (NoOfNonzeroChars == 0)
325  {
326  NoOfZeros++;
327  }
328  else
329  {
330  NoOfNonzeroChars++;
331  }
332  }
333  else
334  {
335  NoOfNonzeroChars++;
336  }
337  argint=argint << 4;
338  }
339 
340  CharSent[nibbles] = '\0' ;
341 
342  PercentageFound=FALSE;
343 
344  /*If the user hasnt specified any number along with format display by default
345  word size*/
346  if (NoOfChars == 0)
347  {
348  NoOfChars=nibbles;
349  }
350 
351  /*print the number displaying as many nibbles as the user has given in print format*/
352  if (NoOfChars >= NoOfNonzeroChars)
353  {
354  for (i=0; i < (NoOfChars - NoOfNonzeroChars); i++)
355  {
356  CON_putchar('0');
357  }
358 
359  for (i=NoOfNonzeroChars; i > 0; i--)
360  {
361  CON_putchar(CharSent[i-1]);
362  }
363  }
364  else
365  {
366  for (i=NoOfNonzeroChars; i > 0; i--)
367  {
368  CON_putchar(CharSent[i-1]);
369  }
370  }
371  }
372 
373  /*if d/x wasnt preceeded by % it is not a format but a string;so just print it*/
374  else
375  {
376  CON_putchar(*((uint8*)format));
377  }
378  /*clean the number of s8 for next cycle */
379  NoOfChars = 0;
380  /*use lower case by default */
381  upperCase = FALSE;
382  break;
383  case '0':
384 
385  /*if 0 was preceeded by % it represents some format */
386  if (PercentageFound == TRUE)
387  {
388  ZeroFound=TRUE;
389  }
390 
391  /*if 0 wasnt preceeded by % it is not a format but a string;so just print it*/
392  else
393  {
394  CON_putchar(*((uint8*)format));
395  }
396  break;
397  case '1':
398  case '2':
399  case '3':
400  case '4':
401  case '5':
402  case '6':
403  case '7':
404  case '8':
405  case '9':
406 
407  /*if the number was preceeded by % and then a 0 it represents some format */
408  if (ZeroFound == TRUE)
409  {
410  NoOfChars = (uint32) (*((uint8*)format) - NUMCHAR_TO_INTEGER);
411  ZeroFound=FALSE;
412  }
413 
414  /*if the number wasnt preceeded by % it is not a format but a string;so just print it*/
415  else
416  {
417  CON_putchar(*((uint8*)format));
418  }
419  break;
420 
421  case '\n':
422  /* Implicitly print a carriage return before each newline */
423  CON_putchar('\r');
424  CON_putchar(*((uint8*)format));
425  break;
426  /*if the character doesnt represent any of these formats just print it*/
427  default:
428  CON_putchar(*((uint8*)format));
429  break;
430  }
431  format = ((uint8*)format)+1;
432  }
433  va_end(argptr);
434 }
