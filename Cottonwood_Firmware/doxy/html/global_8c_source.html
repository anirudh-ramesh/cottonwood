185 bitArrayCopy(const u8 *src_org, s16 src_offset, s16 src_len,
186  u8 *dst_org, s16 dst_offset)
187 {
188 // static const u8 mask[] =
189 // { 0x55, 0x01, 0x03, 0x07, 0x0f, 0x1f, 0x3f, 0x7f, 0xff };
190 // static const u8 mask_xor[] =
191 // { 0x55, 0xfe, 0xfc, 0xf8, 0xf0, 0xe0, 0xc0, 0x80, 0x00 };
192  static const u8 reverse_mask[] =
193  { 0x55, 0x80, 0xc0, 0xe0, 0xf0, 0xf8, 0xfc, 0xfe, 0xff };
194  static const u8 reverse_mask_xor[] =
195  { 0xff, 0x7f, 0x3f, 0x1f, 0x0f, 0x07, 0x03, 0x01, 0x00 };
196 
197  if (src_len) {
198  const u8 *src;
199  u8 *dst;
200  s16 src_offset_modulo,
201  dst_offset_modulo;
202 
203  src = src_org + (src_offset / CHAR_BIT);
204  dst = dst_org + (dst_offset / CHAR_BIT);
205 
206  src_offset_modulo = src_offset % CHAR_BIT;
207  dst_offset_modulo = dst_offset % CHAR_BIT;
208 
209  if (src_offset_modulo == dst_offset_modulo) {
210  s16 byte_len;
211  s16 src_len_modulo;
212  if (src_offset_modulo) {
213  u8 c;
214 
215  c = reverse_mask_xor[dst_offset_modulo] & *src++;
216 
217  PREPARE_FIRST_COPY();
218  *dst++ |= c;
219  }
220 
221  byte_len = src_len / CHAR_BIT;
222  src_len_modulo = src_len % CHAR_BIT;
223 
224  if (byte_len) {
225  memcpy(dst, src, byte_len);
226  src += byte_len;
227  dst += byte_len;
228  }
229  if (src_len_modulo) {
230  *dst &= reverse_mask_xor[src_len_modulo];
231  *dst |= reverse_mask[src_len_modulo] & *src;
232  }
233  } else {
234  s16 bit_diff_ls,
235  bit_diff_rs;
236  s16 byte_len;
237  s16 src_len_modulo;
238  u8 c;
239  /*
240  * Begin: Line things up on destination.
241  */
242  if (src_offset_modulo > dst_offset_modulo) {
243  bit_diff_ls = src_offset_modulo - dst_offset_modulo;
244  bit_diff_rs = CHAR_BIT - bit_diff_ls;
245 
246  c = *src++ << bit_diff_ls;
247  c |= *src >> bit_diff_rs;
248  c &= reverse_mask_xor[dst_offset_modulo];
249  } else {
250  bit_diff_rs = dst_offset_modulo - src_offset_modulo;
251  bit_diff_ls = CHAR_BIT - bit_diff_rs;
252 
253  c = *src >> bit_diff_rs &
254  reverse_mask_xor[dst_offset_modulo];
255  }
256  PREPARE_FIRST_COPY();
257  *dst++ |= c;
258 
259  /*
260  * Middle: copy with only shifting the source.
261  */
262  byte_len = src_len / CHAR_BIT;
263 
264  while (--byte_len >= 0) {
265  c = *src++ << bit_diff_ls;
266  c |= *src >> bit_diff_rs;
267  *dst++ = c;
268  }
269 
270  /*
271  * End: copy the remaining bits;
272  */
273  src_len_modulo = src_len % CHAR_BIT;
274  if (src_len_modulo) {
275  c = *src++ << bit_diff_ls;
276  c |= *src >> bit_diff_rs;
277  c &= reverse_mask[src_len_modulo];
278 
279  *dst &= reverse_mask_xor[src_len_modulo];
280  *dst |= c;
281  }
282  }
