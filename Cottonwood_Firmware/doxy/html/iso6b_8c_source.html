128 s8 iso6bInventoryRound (Tag* tags, u8 maxtags, u8 mask, u8* filter, u8 startaddress)
129 {
130  u8 found = 0;
131  u8 uid[8];
132  u8 mem[8];
133  s8 status;
134  u8 cnt;
135  u8 max_depth = 10;
136  u8 max_commands = 50;
137  iso6bCmd nextCmd;
138 
139  /* Set the length of the UID (8bytes + CRC) */
140  as399xSingleWrite(AS399X_REG_RXLENGTHUP, 0);
141  as399xSingleWrite(AS399X_REG_RXLENGTHLOW, 0x50);
142 
143  as399xEnterDirectMode();
144  AS399X_ENABLE_SENDER();
145  iso6bSendInitialize();
146  iso6bSendSelect(mask, filter, startaddress);
147  AS399X_ENABLE_RECEIVER(); /* Enable also the receiver and now the Tag should respond */
148  status = iso6bReadFifo(uid, 8);
149  if (ISO6B_ERR_NONE == status) /* only one tag in field */
150  {
151  iso6bRead(uid, 0x0, mem, 8);
152  /* we received the data from the tag, the tag is now in DATA_EXCHANGE state
153  and will not continue to participate in the collision arbitration */
154  tags[0].epclen = 8;
155  copyBuffer(uid, tags[0].epc, tags[0].epclen);
156  found = 1;
157  }
158  else if (ISO6B_ERR_NOTAG == status) /* no tags found */
159  {
160  found = 0;
161  }
162  else
163  {
164  /* crc error - two or more tags responded */
165  nextCmd = iso6bSendFail;
166  cnt = 2;
167  while (maxtags && (cnt < max_depth) && max_commands--)
168  {
169  as399xEnterDirectMode();
170  AS399X_ENABLE_SENDER();
171  nextCmd();
172  AS399X_ENABLE_RECEIVER();
173  status = iso6bReadFifo(uid, 8);
174  switch (status)
175  {
176  case ISO6B_ERR_PREAMBLE:
177  nextCmd = iso6bSendFail;
178  cnt ++;
179  break;
180  case ISO6B_ERR_NONE:
181  /* send data_read command so the tag changes its state */
182  iso6bRead(uid, 0x0, mem, 8);
183  /* we received the data from the tag, the tag is now in DATA_EXCHANGE state
184  and will not continue to participate in the collision arbitration */
185  tags[found].epclen = 8;
186  copyBuffer(uid, tags[found].epc, tags[found].epclen);
187  found ++;
188  maxtags --;
189  cnt --;
190  nextCmd = iso6bSendSuccess;
191  break;
192  case ISO6B_ERR_NOTAG:
193  nextCmd = iso6bSendSuccess;
194  cnt --;
195  break;
196  default:
197  if ( nextCmd != iso6bSendResend )
198  {
199  nextCmd = iso6bSendResend;
200  }
201  else
202  {
203  nextCmd = iso6bSendFail;
204  cnt ++;
205  }
206  }
207  }
208  }
209 #if ISO6B_DEBUG
210  {
211  u8 a;
212  CON_print("Number of 6B Tags found: %hhx\n", found);
213  for (a = 0; a < found; a++)
214  {
215  for (cnt = 0; cnt < tags[a].epclen; cnt++)
216  {
217  CON_print("%hhx ",tags[a].epc[cnt]);
218  }
219  CON_print("\n");
220  }
221  CON_print("\n");
222  }
223 #endif
224  if ( found == 0 )
225  { /* From time to time reader ends up in invalid state. */
226  as399xReset();
227  }
228  return found;
229 }

231 s8 iso6bRead(u8* uid, u8 addr, u8* buffer, u8 length)
232 {
233  s8 cnt, i, result;
234  u8 payload[11];
235  u8 cmd = 0x51;
236  u16 crc;
237 
238  length += 2; /* add CRC */
239  /* Set the length of the UID */
240  if (length > 32)
241  {
242  as399xSingleWrite(AS399X_REG_RXLENGTHUP, (length / 32) );
243  }
244  else
245  {
246  as399xSingleWrite(AS399X_REG_RXLENGTHUP, 0);
247  }
248  as399xSingleWrite(AS399X_REG_RXLENGTHLOW, (length % 32) * 8);
249  length -= 2; /* remove CRC again */
250 
251  as399xEnterDirectMode();
252  AS399X_ENABLE_SENDER();
253  IO3(HIGH);
254  mdelay(1);
255  cnt = iso6bPreparePreambleDelimiter(cmdData);
256  cnt += iso6bPrepareByte(&cmdData[cnt], cmd);
257  payload[0] = cmd;
258 
259  for (i = 0; i < 8; i++)
260  {
261  payload[i+1] = uid[i];
262  cnt += iso6bPrepareByte(&cmdData[cnt], payload[i+1]);
263  }
264  payload[9] = addr;
265  cnt += iso6bPrepareByte(&cmdData[cnt], addr);
266  payload[10] = length-1;
267  cnt += iso6bPrepareByte(&cmdData[cnt], length-1);
268 
269  crc = calcCrc16(payload, 11);
270  cnt += iso6bPrepareCRC(&cmdData[cnt], crc);
271 
272  bbRun(cmdData, cnt);
273  IO3(HIGH);
274  AS399X_ENABLE_RECEIVER();
275  result = iso6bReadFifo(buffer, length);
276  if ( result != 0 )
277  { /* From time to time reader ends up in invalid state. */
278  as399xReset();
279  }
280  return result;
281 }

283 s8 iso6bWrite(u8* uid, u8 startaddr, u8* buffer, u8 length)
284 {
285  s8 cnt, i, result;
286  u8 cmd = 0xd; /* command WRITE */
287  u8 payload[11];
288  u8 fifobuffer;
289  u16 crc;
290 
291 #if ISO6B_DEBUG
292  CON_print("iso6bwrite %hhx %hhx %hhx %hhx %hhx %hhx %hhx %hhx: addr=%hhx len=%hhx\n"
293  ,uid[0] ,uid[1] ,uid[2] ,uid[3] ,uid[4] ,uid[5] ,uid[6] ,uid[7]
294  ,startaddr, length);
295 #endif
296  result = ISO6B_ERR_NONE;
297  /* Set size of response (1byte + CRC) */
298  as399xSingleWrite(AS399X_REG_RXLENGTHUP, 0);
299  as399xSingleWrite(AS399X_REG_RXLENGTHLOW, 0x18);
300  while (length -- && (ISO6B_ERR_NONE == result))
301  {
302  as399xEnterDirectMode();
303  AS399X_ENABLE_SENDER();
304  IO3(HIGH);
305  mdelay(1);
306  cnt = iso6bPreparePreambleDelimiter(cmdData);
307  cnt += iso6bPrepareByte(&cmdData[cnt], cmd);
308  payload[0] = cmd;
309 
310  for (i = 0; i < 8; i++)
311  {
312  payload[i+1] = uid[i];
313  cnt += iso6bPrepareByte(&cmdData[cnt], payload[i+1]);
314  }
315  payload[9] = startaddr;
316  cnt += iso6bPrepareByte(&cmdData[cnt], startaddr);
317  startaddr ++;
318  payload[10] = *buffer;
319  cnt += iso6bPrepareByte(&cmdData[cnt], *buffer);
320  buffer ++;
321  crc = calcCrc16(payload, 11);
322  cnt += iso6bPrepareCRC(&cmdData[cnt], crc);
323 
324  bbRun(cmdData, cnt);
325  IO3(HIGH);
326  AS399X_ENABLE_RECEIVER();
327  result = iso6bReadFifo(&fifobuffer, 1);
328  if (ISO6B_ERR_NONE == result)
329  {
330 #if ISO6B_DEBUG
331  CON_print("result = %hhx\n",fifobuffer);
332 #endif
333  result = (fifobuffer == 0) ?
334  ISO6B_ERR_NONE :
335  ISO6B_ERR_NACK;
336  }
337  as399xEnterDirectMode();
338  AS399X_ENABLE_SENDER();
339  IO3(HIGH);
340  /* give the tag some time to write ... */
341  mdelay(IS06B_WRITE_WAIT);
342  /* ... and send a resync */
343  iso6bResync();
344  as399xExitDirectMode();
345  }
346  if ( result != 0 )
347  { /* From time to time reader ends up in invalid state. */
348  as399xReset();
349  }
350  return result;
351 }

376 static s8 iso6bPreparePreambleDelimiter(t_bbData *preData)
377 {
378  /* prepare the preamble and the delimiter1 */
379  /* preamble is 9 times 01 - see section 8.1.4.3 of ISO18000-6b doc */
380  /* delimiter 1 is 1100111010 - see section 8.1.4.4.2 of ISO18000-6b doc */
381  preData[0].bbdata = 0x55; /* 01010101 */
382  preData[1].bbdata = 0x55; /* 01010101 */
383  preData[2].bbdata = 0x73; /* 01 and 110011 */
384  preData[3].bbdata = 0xa; /* 1010 */
385  preData[0].length = 8;
386  preData[1].length = 8;
387  preData[2].length = 8;
388  preData[3].length = 4;
389  return 4;
390 }

434 static s8 iso6bPrepareByte (t_bbData *manchdata, u8 info)
435 {
436  manchdata[0].bbdata = 0;
437  manchdata[1].bbdata = 0;
438  manchdata[0].length = 8;
439  manchdata[1].length = 8;
440 
441  if (info & BIT7) manchdata[0].bbdata |= 2 << 6;
442  else manchdata[0].bbdata |= 1 << 6;
443  if (info & BIT6) manchdata[0].bbdata |= 2 << 4;
444  else manchdata[0].bbdata |= 1 << 4;
445  if (info & BIT5) manchdata[0].bbdata |= 2 << 2;
446  else manchdata[0].bbdata |= 1 << 2;
447  if (info & BIT4) manchdata[0].bbdata |= 2 << 0;
448  else manchdata[0].bbdata |= 1 << 0;
449  if (info & BIT3) manchdata[1].bbdata |= 2 << 6;
450  else manchdata[1].bbdata |= 1 << 6;
451  if (info & BIT2) manchdata[1].bbdata |= 2 << 4;
452  else manchdata[1].bbdata |= 1 << 4;
453  if (info & BIT1) manchdata[1].bbdata |= 2 << 2;
454  else manchdata[1].bbdata |= 1 << 2;
455  if (info & BIT0) manchdata[1].bbdata |= 2 << 0;
456  else manchdata[1].bbdata |= 1 << 0;
457  return 2;
458 }

477 static s8 iso6bPrepareCRC(t_bbData *crcData, s16 crc)
478 {
479  /* send inverted crc (ISO18000-6 spec, chapter 6.5.7.3) */
480  crc = ~crc;
481  iso6bPrepareByte (crcData, (0xff00 & crc) >> 8);
482  crcData += 2;
483  iso6bPrepareByte (crcData, 0xff & crc);
484  return 4; /* return number of used bytes */
485 }

603 static void iso6bSendSelect(u8 mask, u8* filter, u8 startaddress)
604 {
605  u8 bsData[11];
606  s8 cnt;
607  s8 i;
608  u16 crc;
609 
610  cnt = iso6bPreparePreambleDelimiter(cmdData);
611  /* issue GROUP_SELECT_EQ command (0x0) */
612  bsData[0] = 0;
613  cnt += iso6bPrepareByte(&cmdData[cnt], bsData[0]);
614  /* address */
615  bsData[1] = startaddress;
616  cnt += iso6bPrepareByte(&cmdData[cnt], bsData[1]);
617  /* mask - if mask is 0 all tags are selected */
618  bsData[2] = mask;
619  cnt += iso6bPrepareByte(&cmdData[cnt], bsData[2]);
620  /* data */
621  for (i = 3; i < 11; i++)
622  {
623  bsData[i] = *filter++;
624  cnt += iso6bPrepareByte(&cmdData[cnt], bsData[i]);
625  }
626 
627  crc = calcCrc16(bsData, 11);
628  cnt += iso6bPrepareCRC(&cmdData[cnt], crc);
629  bbRun(cmdData, cnt);
630
