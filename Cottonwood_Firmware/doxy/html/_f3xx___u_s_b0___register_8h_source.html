147 /* Register read/write macros */

149 /* These first two macros do not poll for busy, and can be used to increase */
150 /* program speed where neccessary, but should never be used for successive */
151 /* reads or writes. */
152 #define READ_BYTE(addr, target) USB0ADR = (0x80 | addr); \
153  while (USB0ADR & 0x80); target = USB0DAT
154 #define WRITE_BYTE(addr, data) USB0ADR = (addr); USB0DAT = data
155 
156 /* These two macros are polling versions of the above macros, and can be */
157 /* used for successive reads/writes without taking the chance that the */
158 /* Interface Engine is busy from the last register access. */
159 #define POLL_READ_BYTE(addr, target) while(USB0ADR & 0x80); \
160  READ_BYTE(addr, target);
161 #define POLL_WRITE_BYTE(addr, data) while(USB0ADR & 0x80); \
162  WRITE_BYTE(addr, data);
