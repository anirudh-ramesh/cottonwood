75 void initInterface(void)
76 {
77 
78 #if ROGER || MICROREADER
79  P0MDOUT |= 0x04; /* Set P0_2 (DCDC enable) to output */
80  P3MDOUT |= 0x01; /* Set P3_0 (PA enable) to output */
81  DCDC(1); /* Disable DCDC */
82  EN_PA(0); /* Disable PA */
83 #endif
84 #if ROLAND
85  P0MDOUT = 0xd5; /* IO4,7 and CLK output, IRQ, IO6, uart rx, gpio1 input */
86  P0 = (~P0MDOUT) | 0xc0; /* enable LDO and DCDC */
87  P1MDOUT = 0xfc;
88  P1 = (~P1MDOUT) | 0xe4; /* set SW_ANT_P2, SW_RSSI, SEN_TUNE_3, SEN_TUNE_2 */
89  P2MDOUT = 0x06;
90  P2 = (~P2MDOUT) | 0x6;
91  P3MDOUT = 0x04;
92  P3 = (~P3MDOUT);
93  P4MDOUT = 0x06;
94  P4 = (~P4MDOUT);
95 #endif
96 #if ARNIE
97  P0MDOUT = 0x55; /* IO4,7, LED and CLK output; IRQ, IO6, uart rx, clksys input */
98  P0 = (~P0MDOUT) | 0x40;
99  P1MDOUT = 0xFF; /* all outputs */
100  P1 = (~P1MDOUT) | 0x43; /* disable tune caps and switch on antenna P1 */
101  P2MDOUT = 0xE7; /* all outputs except IO5 */
102  P2 = (~P2MDOUT) | 0x16; /* enable PA_LDO and VDD_IO, set IO5 as input */
103  P3MDOUT = 0x0B;
104  P3 = (~P3MDOUT) | 0x01; /* IO1=1, IO0=0, enable of AS399x set to high later (as399xInitialize) */
105  P4MDOUT = 0x00;
106  P4 = (~P4MDOUT);
107 #endif
108 
109  SPI0CFG = 0x60;/* MSTEN=1, CPHA=1, CKPOL=0, */
110 #if (CLK == 48000000) /* reg = f_sys/(2f) - 1 */
111  SPI0CKR = 48/(2*2)-1; /* 2 MHz */
112 #elif (CLK == 24000000)
113  SPI0CKR = 24/(2*2)-1; /* 2 MHz */
114 #elif (CLK == 12000000)
115  SPI0CKR = 12/(2*2)-1; /* 2 MHz */
116 #endif
117  SPI0CN = 1; /* 3-wire, enable SPI */
118  initExtInterrupt();
119 
120 #if (LEO || PICO || ROGER || MICROREADER)
121  P0MDOUT |= 0x05; /* Push pull for MOSI and SCK */
122  P2MDOUT |= 0x06; /*Push Pull for Enable and DEBUG PIN */
123  DPORTDIR(0x1F); /*Push Pull for complete data bus, excluding IO5, IO6 and IO7 */
124  /* Writing to open-drain port a 1 enables high-impedance! */
125  P2_0 = 1; /* CLK pin high-impedance */
126 #endif
127 
128  XBR0 |= 0x2; /* enable SPI interface */
129  XBR1 |= 0xC0; /* enable the crossbar, disable weak pull ups */
130 
131 #if (LEO || PICO || ROGER || MICROREADER)
132  P1_0 = 0; /* Set IO0 to vss and */
133  P1_1 = 1; /* IO1 to vdd to force to SPI mode with following enable toggling */
134 #endif

136

153 void writeReadAS399x( const u8* wbuf, u8 wlen, u8* rbuf, u8 rlen, u8 stopMode, u8 doStart )
154 {
155  u8 i;
156  u8 d;
157 
158  if ( doStart) IO4(0);
159 
160  for ( i = 0; i< wlen; i++ )
161  {
162  SPI0DAT = wbuf[i];
163  while(!(SPI0CN & 0x80));
164  SPI0CN = 1; /* Clear the interrupt flag */
165  d = SPI0DAT; /* dummy receive this byte */
166  }
167 
168  for ( i = 0; i< rlen; i++ )
169  {
170  SPI0DAT = 0;
171  while(!(SPI0CN & 0x80));
172  SPI0CN = 1; /* Clear the interrupt flag */
173  rbuf[i] = SPI0DAT; /* receive this byte */
174  }
175 
176  if ( stopMode != STOP_NONE) IO4(1);
177 }
