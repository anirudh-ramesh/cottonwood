116  P0MDOUT |= 0x04; /* Set P0_2 (DCDC enable) to output */
117  P3MDOUT |= 0x01; /* Set P3_0 (PA enable) to output */

122  initExtInterrupt();
123  SETOUTPORT(LOW); /*all bits high or low while rising edge on EN */
124  CLOCK(LOW);
125  EN(HIGH); /*Enable AS3990 Interface */

129 void writeReadAS399x( const u8* wbuf, u8 wlen, u8* rbuf, u8 rlen, u8 stopMode, u8 doStart )
130 {
131  if (wlen)
132  {
133  DPORTDIRWR(); /*Changes Portdirection to Write */
134  if (doStart) STARTCONDITION();
135  while (wlen--)
136  {
137  SETOUTPORT(*wbuf); /*writes address */
138  NOP();
139  CLOCK(HIGH);
140  NOP();
141  NOP();
142  NOP();
143  CLOCK(LOW);
144  wbuf++;
145  }
146  }
147  if (rlen)
148  {
149  DPORTDIRRD(); /*Changes Portdirection to Read */
150  while (rlen--)
151  {
152  SETOUTPORT(0xFF);
153  CLOCK(HIGH); /*read data */
154  NOP(); /*No Operation */
155  *rbuf = GETPORT();
156  CLOCK(LOW);
157  rbuf++;
158  }
159  }
160  DPORTDIRWR();
161  if ( stopMode == STOP_CONT) STOPCONT(); /*Stopcondition in Continous Mode */
162  if ( stopMode == STOP_SGL ) STOPSGL(); /*Stopcondition in Single Mode */
163 }

166 void writeReadAS399xIsr( const u8 wbuf, u8* rbuf )
167 {
168  /* The NOPDELAY() macros should originally provide a convenient way of producing a
169  * few nops in a sequence. But actually the macros produce much a bigger delay, because
170  * the loop index is put into XDATA by the compiler and the fetching/storing from
171  * XDATA produces quite some overhead. Nevertheless we keep the macros as the
172  * parallel interface needs some delays and I do not want to put huge sequences
173  * of NOP() into the function.
174  */
175  NOPDELAY(1);
176  DPORTDIRWR(); /*Changes Portdirection to Write */
177  NOPDELAY(1);
178  STARTCONDITION();
179  NOPDELAY(1);
180  SETOUTPORT(wbuf); /*writes address */
181  NOPDELAY(1);
182  CLOCK(HIGH);
183  NOPDELAY(1);
184  CLOCK(LOW);
185  NOPDELAY(1);
186 
187  NOPDELAY(1);
188  DPORTDIRRD(); /*Changes Portdirection to Read */
189  NOPDELAY(1);
190  SETOUTPORT(0xFF);
191  CLOCK(HIGH); /*read data */
192  NOPDELAY(1); /*No Operation */
193  *rbuf = GETPORT();
194  CLOCK(LOW);
195  NOPDELAY(1);
196 
197  DPORTDIRWR();
198  NOPDELAY(1);
199  STOPSGL(); /*Stopcondition in Single Mode */
200 }
