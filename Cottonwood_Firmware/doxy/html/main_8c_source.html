97 static void morse_nibble( u8 nibble )
98 {
99  u8 i, cod, code_len;
100 
101  if (nibble > 0xf) nibble = 0x89; /* -..- = "x" */
102 
103  cod = codes[nibble];
104  code_len = cod >> 5;
105 
106  for ( i = code_len; i>0; i-- )
107  {
108  LED1(0);
109  if ( cod & (1<<(i-1)) )
110  {
111  CON_print("-");
112  mdelay(450);
113  }
114  else
115  {
116  CON_print(".");
117  mdelay(150);
118  }
119  LED1(1);
120  mdelay(150);
121  }
122  CON_print(" %hhx\n", nibble);
123  mdelay(450);
124 }

132 static u8 splitPowerCheck(void)
133 {
134 #if ARNIE
135  static u8 startUp = 1; // will be cleared after the first time this function is called
136  static u8 as399xPower = 0; // as399x is without power supply
137  u16 failure = 0;
138  u8 ret = 0;
139 
140  if (as399xPower && EXT_SUPPLY) //quick check if everything is ok
141  return ret;
142 
143  if (startUp)
144  {
145  if (EXT_SUPPLY) // we are powering up and AS399x is supplied, proceed with init
146  {
147  mdelay(1);
148  failure = as399xInitialize(866900);
149  if (!failure)
150  as399xPower = 1;
151  }
152  }
153  else
154  {
155  if(!as399xPower) // no power on AS399x, if it returned we reset, otherwise we report error
156  {
157  ret = 0x80; // return error, except power is back than we reset
158  if (EXT_SUPPLY) // AS399x got power again
159  {
160  mdelay(50);
161  RSTSRC = 0x10; // reset the system
162  }
163  }
164  else if (!EXT_SUPPLY) // AS399x was powered but is not supplied anymore
165  {
166  EN(LOW);
167  VDD_IO = 0;
168  as399xPower = 0;
169  ret = 0x80;
170  }
171  }
172 
173  // if error, display it
174  if (failure)
175  {
176  ret = failure;
177  mdelay(1000);
178  morse_nibble(failure);
179  mdelay(1000);
180 
181 #if ! UARTSUPPORT
182  CON_print("as399xInitialize() returned %x\n", failure);
183 #endif
184  }
185  if (startUp)
186  CON_print("as399xInitialize() returned %x\n", failure);
187 
188  startUp = 0;
189  return ret;
190 #else
191  return 0;
192 #endif
193 }
