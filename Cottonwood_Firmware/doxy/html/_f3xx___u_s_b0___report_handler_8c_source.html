214 // ----------------------------------------------------------------------------
215 // ReportHandler_IN...
216 // ----------------------------------------------------------------------------
217 //
218 // Return Value - None
219 // Parameters - Report ID
220 //
221 // These functions match the Report ID passed as a parameter
222 // to an Input Report Handler.
223 // the ...FG function is called in the SendPacket foreground routine,
224 // while the ...ISR function is called inside the USB ISR. A lock
225 // is set whenever one function is called to prevent a call from the
226 // other from disrupting the routine.
227 // However, this should never occur, as interrupts are disabled by SendPacket
228 // before USB operation begins.
229 // ----------------------------------------------------------------------------
230 void ReportHandler_IN_ISR(unsigned char R_ID)
231 {
232 
233  unsigned char index;
234 
235  index = 0;
236 
237  while (index <= IN_VECTORTABLESize)
238  {
239  // Check to see if Report ID passed into function
240  // matches the Report ID for this entry in the Vector Table
241  if (IN_VECTORTABLE[index].ReportID == R_ID)
242  {
243  IN_VECTORTABLE[index].hdlr();
244  break;
245  }
246 
247  // If Report IDs didn't match, increment the index pointer
248  index++;
249  }
250 
251 }
